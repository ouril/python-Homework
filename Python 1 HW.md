#Задача-1: Дано произвольное целое число, вывести поочередно цифры исходного числа
number = str(input("Введите число"))

range = len(number)-1

sub = int(number)
if sub < 10:

    print(sub)
    
else:

    while sub > 0:
    
        a = sub // (10 ** range)
        
        sub -= a * (10 ** range)
        
        range -= 1
        
        print(a)
